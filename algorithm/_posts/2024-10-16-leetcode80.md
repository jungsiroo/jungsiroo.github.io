<pre><code class="python">class Solution:
    def removeDuplicates(self, nums: List[int]) -> int:
        # Naive Solution
        num_cnt = dict()
        wIdx = 0

        for i in range(len(nums)):
            cnt = num_cnt.get(nums[i], 0)
            if cnt < 2:
                nums[wIdx] = nums[i]
                num_cnt[nums[i]] = cnt+1
                wIdx += 1
        
        return wIdx
</code></pre>
